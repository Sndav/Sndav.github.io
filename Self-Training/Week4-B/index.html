    那我们怎么恢复历史版本呢？？？？

    文件（文件夹） 类型 内容 作业
    config 文本文件 见上图
    HEAD 文件夹 ref: refs/heads/master 当前分支，即 git branch 命令显示的分支
    refs/heads 文件夹 - 本地库所有的分支
    refs/heads/master 文本文件 6975b… master 分支最近一次 commit 的 SHA1 值
    refs/heads/v1 文本文件 fd70… v1 分支最近一次 commit 的 SHA1 值
    refs/remotes 文件夹 -
    refs/remotes/origin 文件夹
    refs/remotes/origin/master 文本文件 6975b…
    refs/remotes/origin/v1 文本文件 e01sd…
    index 文本文件 - 该目录某种程度就是缓冲区（staging area），内容包括它指向的文件的时间戳、文件名、sha1等。
    COMMIT_EDITMSG 文本文件 “some commit description” 最后一次 commit 的注释
    ORIG_HEAD 文本文件 SHA1值
    FETCH_HEAD 文本文件
    logs 文件夹 02b7f... 保存所有更新的引用记录

    这里需要zlib解压

    这里现场写了个python

    #coding: utf-8
import zlib
import argparse
import binascii
from hexdump import hexdump
def compressor(infile, dst, level=6):
    d = ""
    infile = open(infile, "rb")
    dst = open(dst, "wb")
    compress = zlib.compressobj(level)
    data = infile.read(128)
    while data:
        d += compress.compress(data)
        dst.write(compress.compress(data))
        data = infile.read(128)
        dst.write(compress.flush())
    hexdump(d)
    infile.close()
    dst.close()
def decompressor(infile, dst):
    d = ""
    infile = open(infile, "rb")
    dst = open(dst, "wb")
    decompress = zlib.decompressobj()
    data = infile.read(128)
    while data:
        t = decompress.decompress(data)
        dst.write(t)
        d += t
        data = infile.read(128)
        t = decompress.flush()
        dst.write(t) 
        d += t
    hexdump(d)
    infile.close()
    dst.close()
if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("-f", "--file-path", help="file path",required=True)
    parser.add_argument("-o", "--output-path", help="output path",default="file.bin")
    parser.add_argument("-m", "--model", help="model[1 for depress(default) , 0 for compress]",default="1")
    args = parser.parse_args()
    filepath = args.file_path
    output = args.output_path
    model = args.model
    if model == "1":
        decompressor(filepath,output)
    else:
        compressor(filepath,output)
    print("[OK]")
